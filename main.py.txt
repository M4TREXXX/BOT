def load_db():
    data = collection.find_one({"_id": "main_store"})
    if not data:
        data = {"_id": "main_store", "products": {}, "stats": {"stars": 0, "sold": 0}, "users_list": {}}
        collection.insert_one(data)
    return data

def save_db(db):
    collection.replace_one({"_id": "main_store"}, db)

# --- STARS LOGIC ---
async def precheckout_callback(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.pre_checkout_query.answer(ok=True)

async def success_payment_callback(update: Update, context: ContextTypes.DEFAULT_TYPE):
    pay = update.message.successful_payment; p_name = pay.invoice_payload.replace('pay_', ''); db = load_db()
    if p_name in db['products'] and db['products'][p_name]['keys']:
        key = db['products'][p_name]['keys'].pop(0)
        db['stats']['stars'] += pay.total_amount; save_db(db)
        await update.message.reply_text(f"✅ **Success!**\n🎁 {p_name}\n🔑 Key: `{key}`", parse_mode='Markdown')

async def handle_inputs(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.effective_user.id != ADMIN_ID: return
    st = context.user_data.get('st'); db = load_db()
    if st == 'wait_name':
        context.user_data['n'] = update.message.text; context.user_data['st'] = 'wait_price'
        await update.message.reply_text("2️⃣ Price (Stars):")
    elif st == 'wait_price':
        db['products'][context.user_data['n']] = {"price": int(update.message.text), "keys": []}
        save_db(db); context.user_data.clear(); await update.message.reply_text("✅ Product Created!")

def main():
    app = Application.builder().token(TOKEN).build()
    app.add_handler(PreCheckoutQueryHandler(precheckout_callback))
    app.add_handler(MessageHandler(filters.SUCCESSFUL_PAYMENT, success_payment_callback))
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CallbackQueryHandler(handle_callback))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_inputs))
    keep_alive()
    app.run_polling()

if __name__ == '__main__': main()
